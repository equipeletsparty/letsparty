create schema letsplay;

create table letsplay.usuario{
	cpf integer,
	nome varchar,
	username varchar,
	password varchar,
	primary key(cpf)
};

create table letsplay.evento{
	id integer,
	nome varchar,
	descricao text,
	data timestamp,
	longetude real,
	latitude real,
	vaquinha double,
	meta_vaquinha double,
	primary key(id)
};

create table letsplay.article{
	id integer,
	texto text,
	dono varchar,
	data timestamp,
	id_evento integer,
	primary key(id),
	foreign key (id_evento) references letsplay.evento(id)
};

create table letsplay.chatroom{
	id integer,
	nome varchar,
	descricao text,
	primary key(id)
};

create table letsplay.mensagem{
	id integer,
	texto text,
	data timestamp,
	id_chat integer,
	primary key(id),
	foreign key (id_chat) references letsplay.chatroom(id)
};

create table letsplay.participar{
	cpf integer,
	id_evento integer,
	data date,
	primary key(cpf, id_evento),
	foreign key (cpf) references letsplay.usuario(cpf),
	foreign key (id_evento) references letsplay.evento(id) 
};

create table letsplay.join{
	cpf integer,
	id_chat integer,
	primary key(cpf, id_chat),
	foreign key (cpf) references letsplay.usuario(cpf),
	foreign key (id_chat) references letsplay.chatroom(id) 
};
